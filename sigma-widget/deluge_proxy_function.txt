string standalone.api_proxy(string method, string path, string body, string tenant_id) {
	BACKEND_URL = "https://noncorrespondingly-tractile-ava.ngrok-free.dev";
	API_KEY = "ext_key_default";
	API_SECRET = "ext_secret_change_me_in_production";
	
	// Validate input
	if(method.isNull() || path.isNull() || tenant_id.isNull()) {
		return '{"code":"error","message":"Missing required parameters"}';
	}
	
	// Generate timestamp and nonce (Unix seconds)
	// Get current time in milliseconds and convert to seconds
	timestamp_unix = (zoho.currenttime.toTime() / 1000).toLong();
	timestamp_str = timestamp_unix.toString();
	// Remove any decimal point if present
	if(timestamp_str.contains(".")) {
		timestamp_str = timestamp_str.subString(0,timestamp_str.indexOf("."));
	}
	nonce = timestamp_str + "-" + randomNumber(100000,999999) + "-" + randomNumber(100000,999999);
	
	// Build canonical string for signature
	// Format: {timestamp}.{nonce}.{method}.{path}.{body_hash}
	if(body.isNull() || body == "") {
		body_str = "";
	} else {
		body_str = body;
	}
	body_hash = zoho.encryption.sha256(body_str).toLowerCase();
	canonical_string = timestamp_str + "." + nonce + "." + method.toUpperCase() + "." + path + "." + body_hash;
	
	// Compute HMAC-SHA256 signature (keep base64 as-is, don't lowercase!)
	signature = zoho.encryption.hmacsha256(API_SECRET,canonical_string);
	
	// Build headers
	headers = Map();
	headers.put("X-Ext-Key",API_KEY);
	headers.put("X-Ext-Timestamp",timestamp_str);
	headers.put("X-Ext-Nonce",nonce);
	headers.put("X-Ext-Signature",signature);
	headers.put("X-Tenant-ID",tenant_id);
	headers.put("Content-Type","application/json");
	
	// Build full URL
	full_url = BACKEND_URL + path;
	
	// Make HTTP request
	try {
		if(method.equalsIgnoreCase("GET")) {
			response = invokeurl[
				url :full_url
				type :GET
				headers:headers
			];
		}
		else if(method.equalsIgnoreCase("POST")) {
			response = invokeurl[
				url :full_url
				type :POST
				parameters:body_str
				headers:headers
			];
		}
		else if(method.equalsIgnoreCase("PUT")) {
			response = invokeurl[
				url :full_url
				type :PUT
				parameters:body_str
				headers:headers
			];
		}
		else {
			return '{"code":"error","message":"Unsupported HTTP method: ' + method + '"}';
		}
		// Return successful response
		result = Map();
		result.put("code","success");
		details = Map();
		details.put("output",response.toString());
		result.put("details",details);
		return result.toString();
	}
	catch (e) {
		return '{"code":"error","message":"API call failed: ' + e.toString() + '"}';
	}
	return '{"code":"error","message":"Unknown error"}';
}
